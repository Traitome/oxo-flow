# oxo-flow v0.8 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement deterministic core APIs, domain-driven refactoring, diagnostics engine, AI translation layer, collaboration primitives, and three deployment modes for oxo-flow v0.8.

**Architecture:** Domain-driven modular monolith — `domains/{workflow,execution,ai,collaboration,auth,observability}/` with pure service layer + thin HTTP handlers, `infra/{db,hpc,sse,license,rate_limit}/` for infrastructure. StorageBackend trait abstracts SQLite (default) and PostgreSQL (optional).

**Tech Stack:** Rust 2024, axum, sqlx (sqlite + postgres feature), tokio, oxo_flow_core, React 19 + TypeScript + Tailwind CSS + Cytoscape.js (frontend)

**Spec:** `docs/superpowers/specs/2026-06-13-oxo-flow-v0.8-architecture-design.md`

// crates/oxo-flow-web/src/infra/db/sqlite.rs
use sqlx::sqlite::{SqlitePool, SqlitePoolOptions};
use super::{StorageBackend, Pagination, Paginated};
use super::models::*;

pub struct SqliteBackend {
    pool: SqlitePool,
}

impl SqliteBackend {
    pub async fn new(database_url: &str) -> Result<Self, String> {
        let pool = SqlitePoolOptions::new()
            .max_connections(5)
            .connect(database_url)
            .await
            .map_err(|e| e.to_string())?;
        Ok(Self { pool })
    }
}

#[async_trait::async_trait]
impl StorageBackend for SqliteBackend {
    async fn init(&self) -> Result<(), String> {
        sqlx::query("PRAGMA journal_mode=WAL;").execute(&self.pool).await.map_err(|e| e.to_string())?;
        sqlx::query("PRAGMA foreign_keys=ON;").execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS users (
                id TEXT PRIMARY KEY,
                username TEXT UNIQUE NOT NULL,
                role TEXT NOT NULL DEFAULT 'user',
                auth_type TEXT NOT NULL DEFAULT 'password',
                os_user TEXT,
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS pipelines (
                id TEXT PRIMARY KEY,
                user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
                name TEXT NOT NULL,
                version TEXT NOT NULL DEFAULT '0.1.0',
                toml_content TEXT NOT NULL,
                rules_count INTEGER NOT NULL DEFAULT 0,
                forked_from TEXT,
                visibility TEXT NOT NULL DEFAULT 'private',
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
                updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS runs (
                id TEXT PRIMARY KEY,
                user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
                pipeline_id TEXT NOT NULL,
                pipeline_snapshot TEXT NOT NULL,
                status TEXT NOT NULL DEFAULT 'queued',
                phase TEXT NOT NULL DEFAULT 'parsing',
                pid INTEGER,
                workdir TEXT,
                started_at DATETIME,
                finished_at DATETIME
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS run_nodes (
                run_id TEXT NOT NULL REFERENCES runs(id) ON DELETE CASCADE,
                rule_name TEXT NOT NULL,
                status TEXT NOT NULL DEFAULT 'pending',
                started_at DATETIME,
                finished_at DATETIME,
                exit_code INTEGER,
                attempt INTEGER NOT NULL DEFAULT 1,
                error_pattern TEXT,
                PRIMARY KEY (run_id, rule_name)
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS sessions (
                token TEXT PRIMARY KEY,
                user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
                expires_at DATETIME NOT NULL
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS templates (
                id TEXT PRIMARY KEY,
                name TEXT NOT NULL,
                category TEXT NOT NULL DEFAULT 'general',
                description TEXT NOT NULL DEFAULT '',
                tags TEXT NOT NULL DEFAULT '',
                toml_content TEXT NOT NULL,
                is_system INTEGER NOT NULL DEFAULT 0,
                created_by TEXT,
                usage_count INTEGER NOT NULL DEFAULT 0,
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
                updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(
            "CREATE TABLE IF NOT EXISTS shares (
                id TEXT PRIMARY KEY,
                pipeline_id TEXT NOT NULL REFERENCES pipelines(id) ON DELETE CASCADE,
                owner_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
                token TEXT UNIQUE NOT NULL,
                visibility TEXT NOT NULL DEFAULT 'link',
                expires_at DATETIME,
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )"
        ).execute(&self.pool).await.map_err(|e| e.to_string())?;

        sqlx::query(

                id TEXT PRIMARY KEY,
                user_id TEXT NOT NULL,
                action TEXT NOT NULL,
                target TEXT NOT NULL,
                metadata TEXT,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP

    async fn health(&self) -> Result<bool, String> {
        sqlx::query("SELECT 1").execute(&self.pool).await.map(|_| true).map_err(|e| e.to_string())
    }

    async fn create_user(&self, username: &str, role: &str) -> Result<UserRow, String> {
        let id = uuid::Uuid::new_v4().to_string();
        sqlx::query_as::<_, UserRow>(
            "INSERT INTO users (id, username, role, auth_type) VALUES (?, ?, ?, 'password') RETURNING *"
        )
        .bind(&id).bind(username).bind(role)
        .fetch_one(&self.pool).await.map_err(|e| e.to_string())
    }

    async fn get_user_by_id(&self, id: &str) -> Result<Option<UserRow>, String> {
        sqlx::query_as::<_, UserRow>("SELECT * FROM users WHERE id = ?")
            .bind(id).fetch_optional(&self.pool).await.map_err(|e| e.to_string())
    }

    async fn get_user_by_username(&self, username: &str) -> Result<Option<UserRow>, String> {

    async fn save_pipeline(&self, p: &PipelineRow) -> Result<PipelineRow, String> {
        sqlx::query_as::<_, PipelineRow>(
            "INSERT INTO pipelines (id, user_id, name, version, toml_content, rules_count, forked_from, visibility)
             VALUES (?, ?, ?, ?, ?, ?, ?, ?)
             ON CONFLICT(id) DO UPDATE SET name=excluded.name, version=excluded.version,
             toml_content=excluded.toml_content, rules_count=excluded.rules_count,
             updated_at=CURRENT_TIMESTAMP
             RETURNING *"
        )
        .bind(&p.id).bind(&p.user_id).bind(&p.name).bind(&p.version)
        .bind(&p.toml_content).bind(p.rules_count).bind(&p.forked_from).bind(&p.visibility)
        .fetch_one(&self.pool).await.map_err(|e| e.to_string())
    }

    async fn get_pipeline(&self, id: &str) -> Result<Option<PipelineRow>, String> {
        sqlx::query_as::<_, PipelineRow>("SELECT * FROM pipelines WHERE id = ?")
            .bind(id).fetch_optional(&self.pool).await.map_err(|e| e.to_string())
    }

    async fn list_pipelines(&self, user_id: &str, pagination: Pagination) -> Result<Paginated<PipelineRow>, String> {
        let offset = (pagination.page.saturating_sub(1)) * pagination.per_page;
        let total: (i64,) = sqlx::query_as("SELECT COUNT(*) FROM pipelines WHERE user_id = ?")
            .bind(user_id).fetch_one(&self.pool).await.map_err(|e| e.to_string())?;

        let items = sqlx::query_as::<_, PipelineRow>(
            "SELECT * FROM pipelines WHERE user_id = ? ORDER BY updated_at DESC LIMIT ? OFFSET ?"
        )
        .bind(user_id).bind(pagination.per_page as i64).bind(offset as i64)
        .fetch_all(&self.pool).await.map_err(|e| e.to_string())?;

        Ok(Paginated {
            total_items: total.0 as u64,
            total_pages: (total.0 as f64 / pagination.per_page as f64).ceil() as u64,
            page: pagination.page,
            per_page: pagination.per_page,
            items,
        })
    }

use oxo_flow_core::WorkflowConfig;
use super::types::*;

pub fn parse_pipeline(toml_content: &str, _format_version: Option<&str>) -> Result<ParseResponse, String> {
    let config = WorkflowConfig::parse(toml_content).map_err(|e| format!("Parse error: {}", e))?;

    let rules: Vec<RuleSummary> = config.rules.iter().map(|r| {
        RuleSummary {
            name: r.name().to_string(),
            inputs: r.input.iter().map(|p| p.to_string()).collect(),
            outputs: r.output.iter().map(|p| p.to_string()).collect(),
            environment: r.environment.as_ref().map(|e| format!("{:?}", e)),
            threads: Some(r.effective_threads()),
        }
    }).collect();

    let stats = oxo_flow_core::format::workflow_stats(&config);

    let dag = oxo_flow_core::WorkflowDag::from_rules(&config.rules)
        .map_err(|e| format!("DAG build error: {}", e))?;

    let parallel_groups = dag.parallel_groups().unwrap_or_default();
    let metrics = dag.metrics().unwrap_or_default();
    let critical_path = dag.critical_path().unwrap_or_default();

    let nodes: Vec<DagJsonNode> = config.rules.iter().map(|r| {
        let env_color = match r.environment.as_ref().map(|e| e.to_string()).unwrap_or_default().as_str() {
            s if s.contains("conda") => "#3B82F6",
            s if s.contains("docker") => "#8B5CF6",
            s if s.contains("singularity") => "#EC4899",
            _ => "#6B7280",
        };
        DagJsonNode {
            id: r.name().to_string(),
            label: r.name().to_string(),
            color: env_color.to_string(),
        }
    }).collect();

    let edges: Vec<DagJsonEdge> = config.rules.iter().flat_map(|r| {
        dag.dependencies(r.name()).unwrap_or_default().into_iter().map(|dep| {
            DagJsonEdge { from: dep, to: r.name().to_string() }
        }).collect::<Vec<_>>()
    }).collect();

    let pipeline_id = uuid::Uuid::new_v4().to_string();

    Ok(ParseResponse {
        pipeline_id,
        name: config.workflow.name.clone(),
        version: config.workflow.version.clone(),
        rules,
        dag: DagJsonResponse {
            nodes,
            edges,
            parallel_groups,
            critical_path,
            metrics: DagMetrics {
                node_count: metrics.node_count,
                edge_count: metrics.edge_count,
                max_depth: metrics.max_depth,
                max_width: metrics.max_width,
                critical_path_length: metrics.critical_path_length,
                parallel_group_count: metrics.parallel_group_count,
            },
        },
        stats: WorkflowStats {
            rule_count: stats.rule_count,
            shell_rules: stats.shell_rules,
            script_rules: stats.script_rules,
            dependency_count: stats.dependency_count,
            parallel_groups: stats.parallel_groups,
            max_depth: stats.max_depth,
            environments: stats.environments,
            total_threads: stats.total_threads,
            wildcard_count: stats.wildcard_count,
            wildcard_names: stats.wildcard_names,
        },
    })
}

pub fn validate_pipeline(pipeline_id: &str, toml_content: &str) -> Result<ValidateResponse, String> {
    let config = WorkflowConfig::parse(toml_content).map_err(|e| format!("Parse error: {}", e))?;
    let validation = oxo_flow_core::format::validate_format(&config);
    let errors: Vec<ValidationError> = validation.diagnostics.iter().map(|d| {
        ValidationError {
            code: d.code.clone(),
            message: d.message.clone(),
            rule: d.rule.clone(),
            suggestion: d.suggestion.clone(),
        }
    }).collect();

    Ok(ValidateResponse {
        valid: !validation.has_errors(),
        errors,
    })
}

pub fn prepare_pipeline(pipeline_id: &str, toml_content: &str, resolve_wildcards: bool, apply_defaults: bool) -> Result<PrepareResponse, String> {
    let mut config = WorkflowConfig::parse(toml_content).map_err(|e| format!("Parse error: {}", e))?;

    if apply_defaults {
        config.apply_defaults();
    }

    let wildcard_count = if resolve_wildcards {
        config.expand_wildcards().map_err(|e| format!("Wildcard expansion: {}", e))?;
        let rules_before = config.rules.len();
        Ok(rules_before)
    } else {
        Ok(config.rules.len())
    }?;

    // Collect environment setup commands
    let env_resolver = oxo_flow_core::environment::EnvironmentResolver::new();
    let mut setup_cmds = Vec::new();
    for rule in &config.rules {
        if let Some(env_spec) = config.resolve_environment(rule) {
            if let Ok(cmd) = env_resolver.setup_command(&env_spec) {
                setup_cmds.push(format!("{}: {}", rule.name(), cmd));
            }
        }
    }

    Ok(PrepareResponse {
        pipeline_id: pipeline_id.to_string(),
        expanded_rules_count: wildcard_count,
        wildcard_combinations: 0, // computed during actual expansion
        environment_setup_cmds: setup_cmds,
    })
}

    pub fn analyze(&self, rule_name: &str, log_output: &str, exit_code: Option<i32>) -> Vec<DiagnosticResult> {
        let mut results = Vec::new();

        for pattern in &self.patterns {
            let exit_match = exit_code.map_or(true, |ec| pattern.exit_codes.is_empty() || pattern.exit_codes.contains(&ec));
            let stderr_match = pattern.stderr_regex.iter().any(|re| re.is_match(log_output));
            let stdout_match = pattern.stdout_regex.iter().any(|re| re.is_match(log_output));

            if exit_match && (stderr_match || stdout_match || !pattern.exit_codes.is_empty()) {
                let relevant_lines: Vec<String> = log_output.lines()
                    .filter(|line| pattern.stderr_regex.iter().any(|re| re.is_match(line))
                        || pattern.stdout_regex.iter().any(|re| re.is_match(line)))
                    .map(|s| s.to_string())
                    .collect();

                results.push(DiagnosticResult {
                    rule: rule_name.to_string(),
                    error_pattern: Some(pattern.id.to_string()),
                    category: pattern.category.clone(),
                    likely_cause: pattern.likely_cause.to_string(),
                    suggestions: vec![pattern.fix_description.unwrap_or("Manual investigation required.").to_string()],
                    auto_fixable: pattern.auto_fixable,
                    fix_action: pattern.fix_description.map(|desc| FixAction {
                        description: desc.to_string(),
                        config_change: pattern.fix_config_path.map(|path| ConfigChange {
                            path: path.to_string(),
                            old_value: "current".to_string(),
                            new_value: pattern.fix_config_value.unwrap_or("increase").to_string(),
                        }),
                        command: None,
                    }),
                    relevant_log_lines: relevant_lines,
                });
            }
        }

        // If no patterns matched but there's a non-zero exit, add a generic failure
        if results.is_empty() && exit_code.unwrap_or(0) != 0 {
            results.push(DiagnosticResult {
                rule: rule_name.to_string(),
                error_pattern: Some("unknown_error".to_string()),
                category: ErrorCategory::Tool,
                likely_cause: format!("The process exited with code {}. Check logs for details.", exit_code.unwrap()),
                suggestions: vec!["Review the full execution log for clues.".to_string(), "Check input file integrity.".to_string()],
                auto_fixable: false,
                fix_action: None,
                relevant_log_lines: log_output.lines().take(20).map(|s| s.to_string()).collect(),
            });
        }

        results
    }
}

pub fn compute_retry_plan(run_nodes: &[RunNode], dag: &WorkflowDag, from_rule: Option<&str>, skip_succeeded: bool) -> Result<RetryResponse, String> {
    let failed_rules: Vec<&str> = run_nodes.iter()
        .filter(|n| n.status == NodeStatus::Failed)
        .map(|n| n.rule_name.as_str())
        .collect();

    // Find all downstream dependencies of failed rules
    let mut will_rerun: Vec<String> = failed_rules.iter().map(|s| s.to_string()).collect();
    for failed in &failed_rules {
        if let Ok(dependents) = dag.dependents(failed) {
            for dep in dependents {
                if !will_rerun.contains(&dep) {
                    will_rerun.push(dep);
                }
            }
        }
    }

    let will_skip: Vec<String> = run_nodes.iter()
        .filter(|n| n.status == NodeStatus::Success && !will_rerun.contains(&n.rule_name))
        .map(|n| n.rule_name.clone())
        .collect();

    Ok(RetryResponse {
        new_run_id: uuid::Uuid::new_v4().to_string(),
        will_rerun,
        will_skip,
    })
}

    }

    let template_routes = Router::new()
        .route("/api/templates", get(workflow::handlers::list_templates))
        .route("/api/templates/{id}", get(workflow::handlers::get_template))
        .route("/api/templates", post(workflow::handlers::save_template))
        .route("/api/templates/{id}", delete(workflow::handlers::delete_template));

    let ai_routes = Router::new()
        .route("/api/ai/translate", post(ai::handlers::translate))
        .route("/api/ai/explain", post(ai::handlers::explain))
        .route("/api/ai/interpret", post(ai::handlers::interpret))
        .route("/api/ai/optimize", post(ai::handlers::optimize))
        .route("/api/ai/config", get(ai::handlers::get_ai_config))
        .route("/api/ai/config", post(ai::handlers::update_ai_config))
        .route("/api/ai/test", post(ai::handlers::test_ai_config));

    let auth_routes = Router::new()
        .route("/api/auth/login", post(auth::handlers::login))
        .route("/api/auth/me", get(auth::handlers::auth_me))
        .route("/api/users", get(auth::handlers::list_users))
        .route("/api/users", post(auth::handlers::create_user))
        .route("/api/users/{id}", delete(auth::handlers::delete_user));

    let obs_routes = Router::new()
        .route("/api/health", get(observability::handlers::health))
        .route("/api/system", get(observability::handlers::system_info))
        .route("/api/metrics", get(observability::handlers::runtime_metrics))
        .route("/api/events", get(observability::handlers::sse_events))
        .route("/api/audit", get(observability::handlers::get_audit_logs));

    let license_routes = Router::new()
        .route("/api/license", get(auth::handlers::license_status))
        .route("/api/license/upload", post(auth::handlers::upload_license));

    // Preserve v0.7 backward compat routes
    let compat_routes = build_compat_routes();

    Router::new()
        .merge(workflow_routes)
        .merge(run_routes)
        .merge(data_routes)
        .merge(template_routes)
        .merge(ai_routes)
        .merge(auth_routes)
        .merge(obs_routes)
        .merge(license_routes)
        .merge(compat_routes)
        .layer(LicenseHeaderLayer)
        .layer(tower_http::cors::CorsLayer::permissive())
}
